class Animal {
    String name;

    public void setName(String name) {
        this.name = name;
    }

    public String getName() {
        return name;
    }
}

class Dog extends Animal {
    double price;

    public void setPrice(double price) {
        this.price = price;
    }

    public double getPrice() {
        return price;
    }
}

public class Main {
    public static void main(String[] args) {
        Animal animal = new Dog(); // Создаём объект Dog, но ссылка типа Animal

        // Приведение типа и изменение атрибутов
        if (animal instanceof Dog) {
            ((Dog) animal).setName("Max");    // Изменяем имя
            ((Dog) animal).setPrice(400.0);  // Устанавливаем цену

            System.out.println("Имя собаки: " + ((Dog) animal).getName());
            System.out.println("Цена собаки: $" + ((Dog) animal).getPrice());
        } else {
            System.out.println("Это не собака!");
        }
    }
}
